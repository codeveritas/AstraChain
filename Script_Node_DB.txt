CREATE TABLE IF NOT EXISTS `Hash` (
  `id_hash` INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL ,
  `value` REAL NOT NULL
);


CREATE TABLE IF NOT EXISTS `Status` (
  `id` INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL ,
  `status` INTEGER NOT NULL
);


CREATE TABLE IF NOT EXISTS `Transactions` (
  `id` INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL ,
  `aName` LONG NOT NULL,
  `aPays` INTEGER NOT NULL,
  `bName` LONG NOT NULL,
  `bPays` INTEGER NOT NULL,
  `Status` INTEGER NOT NULL,
   FOREIGN KEY (Status) REFERENCES Status (id)
);


CREATE TABLE IF NOT EXISTS `BlockContent` (
  `id` INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL ,
  `Number` LONG NOT NULL,
  `ParentHash` INTEGER NOT NULL,
  `Transactions` INTEGER NOT NULL,
  `bPays` INTEGER NOT NULL,
   FOREIGN KEY (ParentHash) REFERENCES Hash (id_hash),
   FOREIGN KEY (Transactions) REFERENCES Transactions (id)
);


CREATE TABLE IF NOT EXISTS `Block` (
  `id` INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL ,
  `BlockContent` INTEGER NOT NULL,
   FOREIGN KEY (BlockContent) REFERENCES BlockContent (id)
);


CREATE TABLE IF NOT EXISTS `Node` (
  `id` INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL ,
  `Block` INTEGER NOT NULL,
   FOREIGN KEY (Block) REFERENCES Block (id)
);


DROP TABLE `Node`;  
DROP TABLE `Block`; 
DROP TABLE `BlockContent`;
DROP TABLE `Transactions`;
DROP TABLE `Status`;
DROP TABLE `Hash`;
